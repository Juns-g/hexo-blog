function debounce(fn, wait = 500, immeidate) {
  let timer = null
  return function (...args) {
    const runNow = !timer && immeidate
    if (timer) {
      clearTimeout(timer)
      return
    }
    timer = setTimeout(() => {
      fn.apply(this, args)
    }, wait)
    runNow && fn.apply(this, args)
  }
}
