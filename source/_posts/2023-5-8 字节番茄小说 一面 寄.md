```JavaScript
Promise.prototype.myAll(promises){
    if(!Array.isArray(promises))throw Error('error')
    let ans=[]
    let count=0;
    return new Promise((resolve,reject)=>{
        promises.forEach((item,i)=>{
            Promise.resolve(item).then(res=>{
                ans[i]=res
                count++;
                if(count===promises.length)return ans
            }).catch(err=>{
                console.log(err)
            })
        })
    })
}
```
